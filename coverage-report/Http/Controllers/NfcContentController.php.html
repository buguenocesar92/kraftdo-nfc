862     */
863    private function assignDefaultNfcRole(User $user): void
864    {
865        try {
866            // Asignar rol "NFC" específico para usuarios registrados vía onboarding
867            $nfcRole = \Spatie\Permission\Models\Role::where('name', 'NFC')->first();

870                $user->assignRole($nfcRole);
871                \Log::info('Rol NFC asignado a usuario', [
872                    'user_id' => $user->id,
873                    'email' => $user->email,
874                    'role' => 'NFC'
875                ]);
876            } else {
877                // Fallback: Si no existe el rol NFC, crear uno temporal
878                \Log::warning('Rol NFC no encontrado, creando fallback');
879                
880                $fallbackRole = \Spatie\Permission\Models\Role::firstOrCreate(['name' => 'NFC']);
881                
882                // Asignar permisos básicos si es un rol nuevo
883                if ($fallbackRole->wasRecentlyCreated) {
884                    $basicPermissions = \Spatie\Permission\Models\Permission::whereIn('name', [
885                        'access_admin_panel'
886                    ])->get();
887                    
888                    if ($basicPermissions->isNotEmpty()) {
889                        $fallbackRole->syncPermissions($basicPermissions);
890                    }
891                }
892                
893                $user->assignRole($fallbackRole);
894            }
895        } catch (\Exception $e) {
